# Database 생성
mysql -u root –p
show databases; 
use mysql;  
create database python_db; # 사용할 데이터베이스 생성
create user 'python'@'%' identified by 'python'; # user도 만듦
grant all privileges on python_db.* to 'python'@'%';  # 모든 접근 권한 허용
flush privileges;  # 권한 바로 반영 
select user, host from user;
exit;

mysql -u python -p
show grants for current_user;
